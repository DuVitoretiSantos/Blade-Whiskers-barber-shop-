CREATE DATABASE BLADEWHISKERS;
USE `BLADEWHISKERS`;

CREATE TABLE USUARIO(
 id_usuario INT AUTO_INCREMENT NOT NULL,
 nome VARCHAR(45),
 email VARCHAR(45),
 senha VARCHAR(45),
 PRIMARY KEY(id_usuario)
);

CREATE TABLE AGENDAMENTO(
 id_agendamento INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
 anoAg INT, 
 mesAg INT,
 dataAg INT,
 horarioAg INT
);

ALTER TABLE USUARIO
ADD COLUMN agendamento_idForeignKey INT,
ADD FOREIGN KEY (agendamento_idForeignKey)
REFERENCES agendamento (id_agendamento);
